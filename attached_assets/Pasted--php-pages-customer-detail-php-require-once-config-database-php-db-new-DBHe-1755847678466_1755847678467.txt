     LEFT JOIN users u1 ON c.assigned_user_id = u1.id
     LEFT JOIN users u2 ON c.secondary_user_id = u2.id
     WHERE c.id = :id",
    ['id' => $customerId]
);

    ['id' => $customerId]
);

// 첨부 파일 조회

<script>
function openHistoryModal() {
    document.getElementById('historyForm').reset();
    new bootstrap.Modal(document.getElementById('historyModal')).show();
}

async function saveHistory() {
    const formData = new FormData(document.getElementById('historyForm'));
    
    const response = await fetch('api/customers.php?action=add_history', {
        method: 'POST',
        body: formData
    });
    
    const result = await response.json();
    if (result.success) {
        alert('상담 기록이 추가되었습니다.');
        location.reload();
    } else {
        alert(result.message || '저장에 실패했습니다.');
    }
}

function changeStatus() {
    const newStatus = prompt('변경할 상태를 입력하세요 (인텍/수수/접수/작업/완료):', '<?php echo $customer['status']; ?>');
    
    if (newStatus && ['인텍', '수수', '접수', '작업', '완료'].includes(newStatus)) {
        const formData = new FormData();
        formData.append('id', '<?php echo $customerId; ?>');
        formData.append('status', newStatus);
        
        fetch('api/customers.php?action=update', {
            method: 'POST',
            body: formData
        })
        .then(response => response.json())
        .then(result => {
            if (result.success) {
                alert('상태가 변경되었습니다.');
                location.reload();
            } else {
                alert(result.message || '변경에 실패했습니다.');
            }
        });
    }
}

function editCustomer(id) {
    // 부모 페이지의 함수 호출
    window.parent.openCustomerModal(id);
}
