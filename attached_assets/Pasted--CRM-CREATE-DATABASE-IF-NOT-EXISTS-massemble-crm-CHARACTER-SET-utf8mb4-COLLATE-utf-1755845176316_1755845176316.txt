-- 마셈블 CRM 데이터베이스 스키마
CREATE DATABASE IF NOT EXISTS massemble_crm CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
USE massemble_crm;

-- 사용자 테이블
CREATE TABLE IF NOT EXISTS users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL,
    name VARCHAR(100) NOT NULL,
    email VARCHAR(100),
    phone VARCHAR(20),
    department VARCHAR(50),
    team VARCHAR(50),
    role ENUM('admin', 'manager', 'counselor') DEFAULT 'counselor',
    is_active BOOLEAN DEFAULT TRUE,
    last_login DATETIME,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 고객 테이블
CREATE TABLE IF NOT EXISTS customers (
    id INT AUTO_INCREMENT PRIMARY KEY,
    -- 기본 정보
    name VARCHAR(100) NOT NULL,
    phone VARCHAR(20) NOT NULL,
    phone2 VARCHAR(20),
    birth_date DATE,
    gender ENUM('M', 'F', 'N') DEFAULT 'N',
    
    -- 주소 정보
    zipcode VARCHAR(10),
    address VARCHAR(255),
    address_detail VARCHAR(255),
    
    -- 채무 정보
    debt_amount DECIMAL(15, 0),  -- 채무금액
    monthly_income DECIMAL(15, 0),  -- 월소득
    job_type VARCHAR(50),  -- 직업구분
    company_name VARCHAR(100),  -- 회사명
    
    -- 상담 정보
    consult_type VARCHAR(50),  -- 상담유형
    consult_path VARCHAR(50),  -- 유입경로
    status VARCHAR(20) DEFAULT '인텍',  -- 상태 (인텍, 수수, 접수, 작업, 완료)
    
    -- 담당자 정보
    assigned_user_id INT,
    secondary_user_id INT,  -- 2차 상담사
    department VARCHAR(50),
    team VARCHAR(50),
    
    -- 메모
    memo TEXT,
    admin_memo TEXT,  -- 관리자 메모
    
    -- 메타 정보
    source VARCHAR(50),  -- DB출처
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    
    FOREIGN KEY (assigned_user_id) REFERENCES users(id) ON DELETE SET NULL,
    FOREIGN KEY (secondary_user_id) REFERENCES users(id) ON DELETE SET NULL,
    INDEX idx_phone (phone),
    INDEX idx_status (status),
    INDEX idx_assigned (assigned_user_id),
    INDEX idx_created (created_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 상담 이력 테이블
CREATE TABLE IF NOT EXISTS consult_history (
    id INT AUTO_INCREMENT PRIMARY KEY,
    customer_id INT NOT NULL,
    user_id INT NOT NULL,
    consult_date DATETIME DEFAULT CURRENT_TIMESTAMP,
    consult_type VARCHAR(50),  -- 전화, 방문, 카톡 등
    content TEXT,
    next_action VARCHAR(255),
    next_date DATE,
    status_before VARCHAR(20),
    status_after VARCHAR(20),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    FOREIGN KEY (customer_id) REFERENCES customers(id) ON DELETE CASCADE,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
    INDEX idx_customer (customer_id),
    INDEX idx_date (consult_date)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 파일 첨부 테이블
CREATE TABLE IF NOT EXISTS attachments (
    id INT AUTO_INCREMENT PRIMARY KEY,
    customer_id INT NOT NULL,
    file_name VARCHAR(255),
    file_path VARCHAR(500),
    file_type VARCHAR(50),
    file_size INT,
    uploaded_by INT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    FOREIGN KEY (customer_id) REFERENCES customers(id) ON DELETE CASCADE,
    FOREIGN KEY (uploaded_by) REFERENCES users(id) ON DELETE SET NULL,
    INDEX idx_customer (customer_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 시스템 설정 테이블
CREATE TABLE IF NOT EXISTS settings (
    id INT AUTO_INCREMENT PRIMARY KEY,
    setting_key VARCHAR(100) UNIQUE NOT NULL,
    setting_value TEXT,
    setting_type VARCHAR(50),
    description VARCHAR(255),
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 활동 로그 테이블
CREATE TABLE IF NOT EXISTS activity_logs (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT,
    action VARCHAR(50),
    target_type VARCHAR(50),  -- customer, user, etc
    target_id INT,
    details TEXT,
    ip_address VARCHAR(45),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE SET NULL,
    INDEX idx_user (user_id),
    INDEX idx_created (created_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
