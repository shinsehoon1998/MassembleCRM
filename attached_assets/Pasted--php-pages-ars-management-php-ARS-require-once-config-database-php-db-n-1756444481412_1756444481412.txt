<script>
let selectedScenario = 'marketing_consent';
let targetCustomers = [];

function openNewCampaignModal() {
    $('#newCampaignModal').modal('show');
    previewTargets();
}

function selectScenario(scenario) {
    selectedScenario = scenario;
    document.querySelectorAll('.scenario-card').forEach(card => {
        card.classList.remove('selected');
    });
    event.currentTarget.classList.add('selected');
}

async function previewTargets() {
    const status = document.getElementById('statusFilter').value;
    const consent = document.getElementById('consentFilter').value;
    
    // API 호출하여 대상 고객 조회
    const response = await fetch(`api/ars_targets.php?status=${status}&consent=${consent}`);
    const data = await response.json();
    
    targetCustomers = data.customers;
    
    document.getElementById('targetPreview').innerHTML = `
        <div class="alert alert-info">
            <strong>발송 대상: ${data.count}명</strong>
            <div class="mt-2">
                <small>예상 비용: ${(data.count * 35).toLocaleString()}원 (건당 35원)</small>
            </div>
        </div>
    `;
}

async function startCampaign() {
    if (targetCustomers.length === 0) {
        alert('발송 대상이 없습니다.');
        return;
    }
    
    if (!confirm(`${targetCustomers.length}명에게 ARS를 발송하시겠습니까?`)) {
        return;
    }
    
    const sendType = document.querySelector('input[name="sendType"]:checked').value;
    const scheduleTime = sendType === 'schedule' ? document.getElementById('scheduleTime').value : null;
    
    const formData = new FormData();
    formData.append('scenario_id', selectedScenario);
    formData.append('customer_ids', JSON.stringify(targetCustomers.map(c => c.id)));
    formData.append('send_time', scheduleTime || '');
    
    try {
        const response = await fetch('api/ars_campaign.php?action=send_marketing_consent', {
            method: 'POST',
            body: formData
        });
        
        const result = await response.json();
        
        if (result.success) {
            alert('캠페인이 시작되었습니다!');
            $('#newCampaignModal').modal('hide');
            location.reload();
        } else {
            alert('오류: ' + result.message);
        }
    } catch (error) {
        alert('시스템 오류가 발생했습니다.');
        console.error(error);
    }
}

async function checkCampaignStatus(messageId) {
    const response = await fetch(`api/ars_campaign.php?action=check_status&message_id=${messageId}`);
    const result = await response.json();
    
    if (result.success) {
        alert('상태가 업데이트되었습니다.');
        location.reload();
    } else {
        alert('상태 확인 실패: ' + result.message);
    }
}

function viewCampaignDetail(campaignId) {
    window.location.href = `index.php?page=ars_detail&id=${campaignId}`;
}

function downloadReport(campaignId) {
    window.open(`api/ars_report.php?campaign_id=${campaignId}`, '_blank');
}
