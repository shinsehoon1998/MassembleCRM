<script>
// 고객 모달 열기
function openCustomerModal(id = null) {
    if (id) {
        // 수정 모드 - API에서 데이터 로드
        fetch(`api/customers.php?action=get&id=${id}`)
            .then(response => response.json())
            .then(data => {
                if (data.success) {
                    const form = document.getElementById('customerForm');
                    Object.keys(data.customer).forEach(key => {
                        if (form[key]) form[key].value = data.customer[key];
                    });
                }
            });
    } else {
        document.getElementById('customerForm').reset();
        document.getElementById('customerId').value = '';
    }
    
    new bootstrap.Modal(document.getElementById('customerModal')).show();
}

// 고객 저장
async function saveCustomer() {
    const formData = new FormData(document.getElementById('customerForm'));
    const id = formData.get('id');
    
    const response = await fetch('api/customers.php?action=' + (id ? 'update' : 'create'), {
        method: 'POST',
        body: formData
    });
    
    const result = await response.json();
    if (result.success) {
        alert('저장되었습니다.');
        location.reload();
    } else {
        alert(result.message || '저장에 실패했습니다.');
    }
}

// 고객 삭제
async function deleteCustomer(id) {
    if (!confirm('정말 삭제하시겠습니까?')) return;
    
    const response = await fetch(`api/customers.php?action=delete&id=${id}`, {
        method: 'POST'
    });
    
    const result = await response.json();
    if (result.success) {
        alert('삭제되었습니다.');
        location.reload();
    } else {
        alert(result.message || '삭제에 실패했습니다.');
    }
}
