/**
 * 초기 스프레드시트 설정
 */
function setupInitialSheet() {
  try {
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);

    // 헤더 설정 헬퍼 함수
    const setHeadersIfNeeded = (sheet, headers, color) => {
      let needsUpdate = false;
      if (sheet.getLastRow() < 1) {
        needsUpdate = true;
      } else {
        const currentHeaders = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
        if (currentHeaders.length < headers.length || currentHeaders[0] !== headers[0] || currentHeaders[headers.length - 1] !== headers[headers.length - 1]) {
           needsUpdate = true;
        }
      }

      if (needsUpdate) {
        Logger.log(`ℹ️ '${sheet.getName()}' 시트의 헤더를 설정/업데이트합니다.`);
        sheet.getRange(1, 1, 1, sheet.getMaxColumns()).clearContent();
        const headerRange = sheet.getRange(1, 1, 1, headers.length);
        headerRange.setValues([headers]);
        headerRange.setFontWeight('bold').setBackground(color).setFontColor('#FFFFFF');
      }
    };

    // 기존 시트 정리
    const existingSheets = ss.getSheets();
    if (existingSheets.length === 1 && existingSheets[0].getName() === 'Sheet1') {
      existingSheets[0].setName(SHEETS.SURVEY);
    }

    // 설문 응답 시트
    let surveySheet = ss.getSheetByName(SHEETS.SURVEY);
    if (!surveySheet) surveySheet = ss.insertSheet(SHEETS.SURVEY, 0);
    const surveyHeaders = [
      '응답시간', 'ID', '이름', '전화번호', '생년월일', '지역',
      'CRM전송상태', 'CRM전송시간', '마셈블고객ID'
    ];
    setHeadersIfNeeded(surveySheet, surveyHeaders, '#4285F4');

    // 분석 결과
    let analysisSheet = ss.getSheetByName(SHEETS.ANALYSIS);
    if (!analysisSheet) analysisSheet = ss.insertSheet(SHEETS.ANALYSIS, 1);
    const analysisHeaders = [
      '분석시간', '설문ID', '점수', '또래평균', '차이', '연령', '연령그룹',
      '추천사항1', '추천사항2', '추천사항3', '추천사항4'
    ];
    setHeadersIfNeeded(analysisSheet, analysisHeaders, '#34A853');

    // 상담 신청
    let consultationSheet = ss.getSheetByName(SHEETS.CONSULTATION);
    if (!consultationSheet) consultationSheet = ss.insertSheet(SHEETS.CONSULTATION, 2);
    const consultationHeaders = ['신청시간', '상담ID', '이름', '전화번호', '희망시간', '개인정보동의', '설문ID'];
    setHeadersIfNeeded(consultationSheet, consultationHeaders, '#FF9900');

    // CRM 전송 로그
    let crmLogSheet = ss.getSheetByName(SHEETS.CRM_LOG);
    if (!crmLogSheet) crmLogSheet = ss.insertSheet(SHEETS.CRM_LOG, 3);
    const crmLogHeaders = ['전송시간', '설문ID', '이름', '전화번호', '상태', 'CRM고객ID', '오류메시지'];
    setHeadersIfNeeded(crmLogSheet, crmLogHeaders, '#9C27B0');

    // 너비 자동
    ss.getSheetByName(SHEETS.SURVEY).autoResizeColumns(1, surveyHeaders.length);
    ss.getSheetByName(SHEETS.ANALYSIS).autoResizeColumns(1, analysisHeaders.length);
    ss.getSheetByName(SHEETS.CONSULTATION).autoResizeColumns(1, consultationHeaders.length);
    ss.getSheetByName(SHEETS.CRM_LOG).autoResizeColumns(1, crmLogHeaders.length);

    Logger.log('✅ 스프레드시트 초기 설정 완료');
    return '스프레드시트 초기 설정 완료';
  } catch (err) {
    Logger.log('❌ 초기 설정 오류: ' + err);
    throw err;
  }
}

/**
 * 웹앱 POST 요청 처리
 */
function doPost(e) {
  try {
    Logger.log('🔥 doPost 호출됨');
    if (!e || !e.postData || !e.postData.contents) {
      return ContentService.createTextOutput(JSON.stringify({success:false,error:'데이터가 없습니다.'}))
        .setMimeType(ContentService.MimeType.JSON);
    }

    const data = JSON.parse(e.postData.contents);
    const { action, payload } = data || {};
    Logger.log('🎯 액션: ' + action);

    let result = { success:false, message:'알 수 없는 액션입니다.' };

    switch (action) {
      case 'survey_complete':      result = handleSurveyComplete(payload); break;
      case 'consultation_request': result = handleConsultationRequest(payload); break;
      case 'get_stats':            result = getStatistics();                break;
      case 'send_crm':             result = sendToMassembleCRM(payload?.surveyId || null); break;
      case 'send_all_crm':         result = sendAllToMassembleCRM();       break;
      case 'check_crm':            result = checkMassembleCRMStatus();     break;
      default:                     result = { success:false, message:'지원하지 않는 액션: ' + action };
    }

    return ContentService.createTextOutput(JSON.stringify(result))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (error) {
    Logger.log('❌ doPost 오류: ' + error + '\n' + error.stack);
    return ContentService.createTextOutput(JSON.stringify({success:false,error:'서버 오류: ' + error.message}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

/**
 * 웹앱 GET 요청 처리
 */
function doGet(e) {
  try {
    const action = e.parameter.action || 'stats';
    let result = {};
    if (action === 'stats')      result = getStatistics();
    else if (action === 'recent')  result = getRecentData(parseInt(e.parameter.limit) || 20);
    else if (action === 'export')  result = exportData();
    else                           result = { error: '알 수 없는 액션: ' + action };

    return ContentService.createTextOutput(JSON.stringify(result, null, 2))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({ error: '조회 오류: ' + error.message }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

/**
 * 설문 완료 저장
 */
function handleSurveyComplete(payload) {
  try {
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    const surveySheet = ss.getSheetByName(SHEETS.SURVEY);
    const analysisSheet = ss.getSheetByName(SHEETS.ANALYSIS);
    if (!surveySheet || !analysisSheet) throw new Error('시트를 찾을 수 없습니다. 먼저 setupInitialSheet() 실행');

    const now = new Date();
    const surveyId = payload.surveyId || generateId();

    const surveyRow = [
      now,
      surveyId,
      payload.name || '',
      payload.phone || '',
      payload.birthDate || '',
      payload.region || '',
      'pending',
      '',
      ''
    ];
    surveySheet.appendRow(surveyRow);

    if (payload.analysis) {
      const r = payload.analysis.recommendations || [];
      const analysisRow = [
        now,
        surveyId,
        payload.analysis.score || 0,
        payload.analysis.peerAverage || 0,
        payload.analysis.difference || 0,
        payload.analysis.age || 0,
        payload.analysis.ageGroup || '',
        r[0] || '',
        r[1] || '',
        r[2] || '',
        r[3] || ''
      ];
      analysisSheet.appendRow(analysisRow);
    }

    const last = surveySheet.getLastRow();
    surveySheet.getRange(last,1).setNumberFormat('yyyy-mm-dd hh:mm:ss');
    
    if (payload.analysis) {
      const la = analysisSheet.getLastRow();
      analysisSheet.getRange(la,1).setNumberFormat('yyyy-mm-dd hh:mm:ss');
      analysisSheet.getRange(la,3,1,3).setNumberFormat('#,##0');
    }

    return { success:true, message:'저장 완료', surveyId, timestamp: now.toISOString() };
  } catch (error) {
    Logger.log('❌ 설문 완료 처리 오류: ' + error);
    return { success:false, error: error.message };
  }
}

/**
 * 마셈블 CRM으로 설문조사 데이터 전송 (행 번호 기준)
 */
function sendToMassembleCRMByRow(rowNumber) {
  if (MASSEMBLE_CRM_URL.length < 20 || !MASSEMBLE_API_KEY) {
    const errorMsg = '마셈블 CRM 주소 또는 API 키가 설정되지 않았습니다.';
    Logger.log(`❌ CONFIG ERROR: ${errorMsg}`);
    try {
        const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
        const surveySheet = ss.getSheetByName(SHEETS.SURVEY);
        if(surveySheet) {
          surveySheet.getRange(rowNumber, 7).setValue('config_error');
        }
    } catch(e) {
      Logger.log(`시트에 오류 상태 기록 실패: ${e}`);
    }
    throw new Error(errorMsg);
  }

  try {
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    const surveySheet = ss.getSheetByName(SHEETS.SURVEY);
    const crmLogSheet = ss.getSheetByName(SHEETS.CRM_LOG);
    
    if (!surveySheet) throw new Error('설문_응답 시트를 찾을 수 없습니다.');

    const rowData = surveySheet.getRange(rowNumber, 1, 1, surveySheet.getLastColumn()).getValues()[0];
    
    const surveyId = rowData[1];
    const name = rowData[2];
    const phone = rowData[3];
    const birthDate = rowData[4];
    const region = rowData[5];
    const crmStatus = rowData[6];

    if (crmStatus === 'sent' || crmStatus === 'success') {
      Logger.log(`⏭️ 행 ${rowNumber}: 이미 CRM 전송 완료 (${surveyId})`);
      return { success: true, message: '이미 전송됨', surveyId };
    }

    if (!surveyId || !name || !phone) {
      Logger.log(`⏭️ 행 ${rowNumber}: 필수 데이터 누락`);
      surveySheet.getRange(rowNumber, 7).setValue('data_missing');
      return { success: false, message: '필수 데이터 누락' };
    }

    const payload = {
      name: name,
      phone: phone,
      birthDate: birthDate,
      region: region,
      source: '보멘토_설문',
      surveyId: surveyId
    };

    const options = {
      method: 'post',
      contentType: 'application/json',
      headers: {
        'Authorization': 'Bearer ' + MASSEMBLE_API_KEY
      },
      payload: JSON.stringify(payload),
      muteHttpExceptions: true
    };

    Logger.log(`📤 CRM 전송 시도: ${name} (${phone})`);
    const response = UrlFetchApp.fetch(MASSEMBLE_CRM_URL + '/api/leads', options);
    const responseCode = response.getResponseCode();
    const responseText = response.getContentText();

    Logger.log(`📥 CRM 응답 코드: ${responseCode}`);
    Logger.log(`📥 CRM 응답 내용: ${responseText}`);

    const now = new Date();

    if (responseCode === 200 || responseCode === 201) {
      let crmCustomerId = '';
      try {
        const responseData = JSON.parse(responseText);
        crmCustomerId = responseData.customerId || responseData.id || '';
      } catch(e) {
        Logger.log('CRM 응답 파싱 실패, 하지만 전송 성공으로 처리');
      }

      surveySheet.getRange(rowNumber, 7).setValue('success');
      surveySheet.getRange(rowNumber, 8).setValue(now);
      surveySheet.getRange(rowNumber, 8).setNumberFormat('yyyy-mm-dd hh:mm:ss');
      if (crmCustomerId) {
        surveySheet.getRange(rowNumber, 9).setValue(crmCustomerId);
      }

      if (crmLogSheet) {
        crmLogSheet.appendRow([now, surveyId, name, phone, 'success', crmCustomerId, '']);
        crmLogSheet.getRange(crmLogSheet.getLastRow(), 1).setNumberFormat('yyyy-mm-dd hh:mm:ss');
      }

      Logger.log(`✅ CRM 전송 성공: ${surveyId}`);
      return { success: true, message: 'CRM 전송 성공', surveyId, crmCustomerId };
    } else {
      const errorMsg = `HTTP ${responseCode}: ${responseText.substring(0, 200)}`;
      surveySheet.getRange(rowNumber, 7).setValue('failed');
      
      if (crmLogSheet) {
        crmLogSheet.appendRow([now, surveyId, name, phone, 'failed', '', errorMsg]);
        crmLogSheet.getRange(crmLogSheet.getLastRow(), 1).setNumberFormat('yyyy-mm-dd hh:mm:ss');
      }

      Logger.log(`❌ CRM 전송 실패: ${errorMsg}`);
      return { success: false, message: 'CRM 전송 실패', error: errorMsg };
    }
  } catch (error) {
    Logger.log(`❌ CRM 전송 오류 (행 ${rowNumber}): ${error}`);
    try {
      const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
      const surveySheet = ss.getSheetByName(SHEETS.SURVEY);
      const crmLogSheet = ss.getSheetByName(SHEETS.CRM_LOG);
      
      if (surveySheet) {
        surveySheet.getRange(rowNumber, 7).setValue('error');
        const rowData = surveySheet.getRange(rowNumber, 1, 1, 6).getValues()[0];
        
        if (crmLogSheet) {
          crmLogSheet.appendRow([new Date(), rowData[1], rowData[2], rowData[3], 'error', '', error.toString()]);
          crmLogSheet.getRange(crmLogSheet.getLastRow(), 1).setNumberFormat('yyyy-mm-dd hh:mm:ss');
        }
      }
    } catch(e2) {
      Logger.log(`오류 로깅 실패: ${e2}`);
    }
    return { success: false, error: error.toString() };
  }
}

/**
 * 특정 설문 ID로 CRM 전송
 */
function sendToMassembleCRM(surveyId) {
  try {
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    const surveySheet = ss.getSheetByName(SHEETS.SURVEY);
    if (!surveySheet) throw new Error('설문_응답 시트를 찾을 수 없습니다.');

    const data = surveySheet.getDataRange().getValues();
    for (let i = 1; i < data.length; i++) {
      if (data[i][1] === surveyId) {
        return sendToMassembleCRMByRow(i + 1);
      }
    }
    return { success: false, message: '설문 ID를 찾을 수 없습니다: ' + surveyId };
  } catch (error) {
    return { success: false, error: error.toString() };
  }
}

/**
 * 모든 pending 상태 데이터를 CRM으로 전송
 */
function sendAllToMassembleCRM() {
  try {
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    const surveySheet = ss.getSheetByName(SHEETS.SURVEY);
    if (!surveySheet) throw new Error('설문_응답 시트를 찾을 수 없습니다.');

    const data = surveySheet.getDataRange().getValues();
    let successCount = 0;
    let failCount = 0;
    let skipCount = 0;

    for (let i = 1; i < data.length; i++) {
      const crmStatus = data[i][6];
      
      if (crmStatus === 'pending' || crmStatus === 'failed' || crmStatus === 'error' || crmStatus === '') {
        const result = sendToMassembleCRMByRow(i + 1);
        if (result.success) {
          successCount++;
        } else {
          failCount++;
        }
        Utilities.sleep(500);
      } else {
        skipCount++;
      }
    }

    return {
      success: true,
      message: `일괄 전송 완료: 성공 ${successCount}, 실패 ${failCount}, 건너뜀 ${skipCount}`
    };
  } catch (error) {
    return { success: false, error: error.toString() };
  }
}

    return result;

    Logger.log('❌ 크론 작업 오류: ' + error);

  }
}

/**
 * 크론 트리거 생성 (1회만 실행)
 */
function createCronTrigger() {
  // 기존 트리거 삭제
  const triggers = ScriptApp.getProjectTriggers();
  triggers.forEach(trigger => {
    if (trigger.getHandlerFunction() === 'cronJob') {
      ScriptApp.deleteTrigger(trigger);
    }
  });

  // 1분마다 실행되는 트리거 생성
  ScriptApp.newTrigger('cronJob')
    .timeBased()
    .everyMinutes(1)
    .create();

  Logger.log('✅ 크론 트리거 생성 완료 (1분 주기)');
  return '크론 트리거 생성 완료';
}

/**
 * 크론 트리거 삭제
 */
function deleteCronTrigger() {
  const triggers = ScriptApp.getProjectTriggers();
  let deletedCount = 0;
  triggers.forEach(trigger => {
    if (trigger.getHandlerFunction() === 'cronJob') {
      ScriptApp.deleteTrigger(trigger);
      deletedCount++;
    }
  });
  Logger.log(`🗑️ ${deletedCount}개의 크론 트리거 삭제 완료`);
  return `${deletedCount}개의 크론 트리거 삭제 완료`;
}

/**
 * 통계 조회
 */
function getStatistics() {
  try {
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    const surveySheet = ss.getSheetByName(SHEETS.SURVEY);
    const consultationSheet = ss.getSheetByName(SHEETS.CONSULTATION);

    const totalSurveys = Math.max(0, surveySheet.getLastRow() - 1);
    const totalConsultations = Math.max(0, consultationSheet.getLastRow() - 1);

    const surveyData = surveySheet.getDataRange().getValues();
    let crmSent = 0;
    let crmPending = 0;
    let crmFailed = 0;

    for (let i = 1; i < surveyData.length; i++) {
      const status = surveyData[i][6];
      if (status === 'success' || status === 'sent') crmSent++;
      else if (status === 'pending' || status === '') crmPending++;
      else if (status === 'failed' || status === 'error') crmFailed++;
    }

    return {
      success: true,
      statistics: {
        totalSurveys,
        totalConsultations,
        crmSent,
        crmPending,
        crmFailed,
        timestamp: new Date().toISOString()
      }
    };
  } catch (error) {
    return { success: false, error: error.toString() };
  }
}

/**
 * 최근 데이터 조회
 */
function getRecentData(limit) {
  try {
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    const surveySheet = ss.getSheetByName(SHEETS.SURVEY);
    
    const lastRow = surveySheet.getLastRow();
    const startRow = Math.max(2, lastRow - limit + 1);
    const numRows = lastRow - startRow + 1;

    if (numRows <= 0) {
      return { success: true, data: [], message: '데이터 없음' };
    }

    const data = surveySheet.getRange(startRow, 1, numRows, surveySheet.getLastColumn()).getValues();
    
    return {
      success: true,
      data: data.map(row => ({
        timestamp: row[0],
        surveyId: row[1],
        name: row[2],
        phone: row[3],
        birthDate: row[4],
        region: row[5],
        crmStatus: row[6],
        crmSentAt: row[7],
        crmCustomerId: row[8]
      }))
    };
  } catch (error) {
    return { success: false, error: error.toString() };
  }
}

/**
 * 전체 데이터 내보내기
 */
function exportData() {
  try {
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    
    return {
      success: true,
      spreadsheetId: SPREADSHEET_ID,
      spreadsheetUrl: ss.getUrl(),
      sheets: {
        survey: SHEETS.SURVEY,
        analysis: SHEETS.ANALYSIS,
        consultation: SHEETS.CONSULTATION,
        crmLog: SHEETS.CRM_LOG
      }
    };
  } catch (error) {
    return { success: false, error: error.toString() };
  }
}

/**
 * ID 생성 헬퍼
 */
function generateId() {
  return 'survey_' + new Date().getTime() + '_' + Math.random().toString(36).substr(2, 9);
}
