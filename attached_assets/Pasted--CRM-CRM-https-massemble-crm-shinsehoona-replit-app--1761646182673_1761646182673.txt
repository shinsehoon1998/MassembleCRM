/**

 * 처음 한 번만 실행하세요
 */
function setupLeadSheet() {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    let leadSheet = ss.getSheetByName(LEAD_SHEET_NAME);
    
    // 시트가 없으면 생성
    if (!leadSheet) {
      leadSheet = ss.insertSheet(LEAD_SHEET_NAME, 0);
    }
    
    // 헤더 설정
    const headers = [
      '등록일시',      // A (1)
      '이름',          // B (2)
      '전화번호',      // C (3)
      '문의사항',      // D (4)
      '출처',          // E (5)
      'CRM전송상태',   // F (6)
      'CRM전송시간',   // G (7)
      '마셈블고객ID'   // H (8)
    ];
    
    // 헤더가 없거나 다르면 설정
    if (leadSheet.getLastRow() < 1 || leadSheet.getRange(1, 1).getValue() !== headers[0]) {
      leadSheet.getRange(1, 1, 1, headers.length).setValues([headers]);
      leadSheet.getRange(1, 1, 1, headers.length)
        .setFontWeight('bold')
        .setBackground('#4285F4')
        .setFontColor('#FFFFFF');
    }
    
    // 열 너비 자동 조정
    leadSheet.autoResizeColumns(1, headers.length);
    
    Logger.log('✅ 리드 시트 초기 설정 완료');
    return '리드 시트 초기 설정 완료';
  } catch (err) {
    Logger.log('❌ 초기 설정 오류: ' + err);
    throw err;
  }
}

/**
 * CRM 전송 함수 (특정 행)
 */
function sendLeadToCRMByRow(rowNumber) {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const leadSheet = ss.getSheetByName(LEAD_SHEET_NAME);
    
    if (!leadSheet) {
      throw new Error('리드데이터 시트를 찾을 수 없습니다. setupLeadSheet() 실행 필요');
    }
    
    const rowData = leadSheet.getRange(rowNumber, 1, 1, leadSheet.getLastColumn()).getValues()[0];
    
    // 컬럼 인덱스 (0-based)
    const IDX = {
      timestamp: 0,    // A
      name: 1,         // B
      phone: 2,        // C
      inquiry: 3,      // D
      source: 4,       // E
      crmStatus: 5,    // F
      crmSentTime: 6,  // G
      customerId: 7    // H
    };
    
    // 이미 전송 성공한 경우 스킵
    const status = String(rowData[IDX.crmStatus] || '').trim().toLowerCase();
    if (status === 'success' || status === 'sending') {
      return { success: false, message: `이미 처리됨: ${status}`, row: rowNumber };
    }
    
    // 전송 중 상태로 변경
    leadSheet.getRange(rowNumber, IDX.crmStatus + 1).setValue('sending');
    
    // 전화번호 정제
    let phone = String(rowData[IDX.phone] || '').replace(/[^0-9]/g, '');
    if (phone.length === 10 && phone.startsWith('10')) {
      phone = '0' + phone;
    }
    
    // 필수 필드 검증
    const customerName = String(rowData[IDX.name] || '').trim();
    const customerPhone = phone || '';
    
    if (!customerName || !customerPhone || customerPhone.length < 10) {
      leadSheet.getRange(rowNumber, IDX.crmStatus + 1).setValue('validation_error');
      return {
        success: false,
        message: '필수 정보 부족 (이름/전화번호)',
        row: rowNumber
      };
    }
    
    // CRM 전송 데이터 구성
    const crmPayload = {
      name: customerName,
      phone: customerPhone,
      consultType: '보험상담',
      consultPath: '광고리드',
      source: rowData[IDX.source] || 'insurance_ad_lead',
      marketingConsent: false,
      
      info1: rowData[IDX.inquiry] || '',
      info2: rowData[IDX.source] || '',
      info10: `[${new Date().toLocaleString('ko-KR')}] 광고 리드 연동`,
      
      memo1: ''
    };
    
    Logger.log('📤 CRM 전송 데이터 (Row ' + rowNumber + '): ' + JSON.stringify(crmPayload, null, 2));
    
    // CRM API 호출
    const response = UrlFetchApp.fetch(CRM_API_URL, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'X-API-Key': CRM_API_KEY
      },
      payload: JSON.stringify(crmPayload),
      muteHttpExceptions: true
    });
    
    const responseText = response.getContentText();
    const statusCode = response.getResponseCode();
    const now = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'yyyy-MM-dd HH:mm:ss');
    
    Logger.log(`📥 CRM 응답 (Row ${rowNumber}, ${statusCode}): ${responseText}`);
    
    if (statusCode === 200) {
      const result = JSON.parse(responseText);
      
      if (result.success) {
        // 성공 상태 업데이트
        leadSheet.getRange(rowNumber, IDX.crmStatus + 1).setValue('success');
        leadSheet.getRange(rowNumber, IDX.crmSentTime + 1).setValue(now);
        leadSheet.getRange(rowNumber, IDX.customerId + 1).setValue(result.customerId || '');
        
        Logger.log(`✅ CRM 전송 성공 (Row ${rowNumber}): ${customerName} → ${result.customerId}`);
        
        return {
          success: true,
          message: 'CRM 전송 성공',
          row: rowNumber,
          customerId: result.customerId,
          name: customerName,
          phone: customerPhone
        };
      } else {
        // API 성공했지만 비즈니스 로직 실패
        const errorMessage = `api_error: ${result.message.substring(0, 100)}`;
        leadSheet.getRange(rowNumber, IDX.crmStatus + 1).setValue(errorMessage);
        leadSheet.getRange(rowNumber, IDX.crmSentTime + 1).setValue(now);
        
        return {
          success: false,
          message: `CRM API 오류: ${result.message}`,
          error: result.message,
          row: rowNumber
        };
      }
    } else {
      // HTTP 오류
      leadSheet.getRange(rowNumber, IDX.crmStatus + 1).setValue(`http_error_${statusCode}`);
      leadSheet.getRange(rowNumber, IDX.crmSentTime + 1).setValue(now);
      
      return {
        success: false,
        message: `CRM HTTP 오류 (${statusCode}): ${responseText}`,
        error: `HTTP ${statusCode}`,
        row: rowNumber
      };
    }
  } catch (error) {
    Logger.log('❌ CRM 전송 오류 (Row ' + rowNumber + '): ' + error);
    
    try {
      const ss = SpreadsheetApp.getActiveSpreadsheet();
      const leadSheet = ss.getSheetByName(LEAD_SHEET_NAME);
      if (leadSheet) {
        leadSheet.getRange(rowNumber, 6).setValue('error: ' + error.toString().substring(0, 50));
      }
    } catch (e) {
      Logger.log('시트 오류 상태 기록 실패: ' + e);
    }
    
    return {
      success: false,
      message: '전송 중 오류: ' + error.message,
      error: error.toString(),
      row: rowNumber
    };
  }
}

/**
 * Cron Job: 1분마다 자동 실행
 * pending 상태인 리드를 CRM으로 전송
 */
function cronJob() {
  try {
    Logger.log('🔄 Cron Job 시작: ' + new Date().toLocaleString('ko-KR'));
    
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const leadSheet = ss.getSheetByName(LEAD_SHEET_NAME);
    
    if (!leadSheet) {
      Logger.log('⚠️ 리드데이터 시트를 찾을 수 없습니다.');
      return;
    }
    
    const lastRow = leadSheet.getLastRow();
    if (lastRow <= 1) {
      Logger.log('ℹ️ 처리할 데이터가 없습니다.');
      return;
    }
    
    // CRM전송상태 열(F열) 전체 읽기
    const statusRange = leadSheet.getRange(2, 6, lastRow - 1, 1);
    const statuses = statusRange.getValues();
    
    let sentCount = 0;
    let errorCount = 0;
    const maxSendPerRun = 50; // 한 번에 최대 50개만 전송
    
    // pending 또는 빈 상태인 행만 전송
    for (let i = 0; i < statuses.length && sentCount < maxSendPerRun; i++) {
      const rowNumber = i + 2; // 헤더 제외하므로 +2
      const status = String(statuses[i][0] || '').trim().toLowerCase();
      
      // pending, 빈값, 또는 http_error로 시작하는 경우만 재전송
      if (status === '' || status === 'pending' || status.startsWith('http_error')) {
        Logger.log(`📨 Row ${rowNumber} 전송 시도 (상태: ${status || '빈값'})`);
        
        const result = sendLeadToCRMByRow(rowNumber);
        
        if (result.success) {
          sentCount++;
          Logger.log(`  ✅ 성공: ${result.name}`);
        } else {
          errorCount++;
          Logger.log(`  ❌ 실패: ${result.message}`);
        }
        
        // API 호출 간 짧은 대기 (과부하 방지)
        Utilities.sleep(200);
      }
    }
    
    Logger.log(`✅ Cron Job 완료: 성공 ${sentCount}건, 오류 ${errorCount}건`);
    
    return {
      success: true,
      sentCount: sentCount,
      errorCount: errorCount,
      timestamp: new Date().toISOString()
    };
    
  } catch (error) {
    Logger.log('❌ Cron Job 오류: ' + error);
    return {
      success: false,
      error: error.toString()
    };
  }
}

/**
 * Cron Trigger 생성 (1분마다 자동 실행)
 * 처음 한 번만 실행하세요
 */
function createCronTrigger() {
  try {
    // 기존 트리거 삭제
    const triggers = ScriptApp.getProjectTriggers();
    triggers.forEach(trigger => {
      if (trigger.getHandlerFunction() === 'cronJob') {
        ScriptApp.deleteTrigger(trigger);
      }
    });
    
    // 1분마다 실행되는 새 트리거 생성
    ScriptApp.newTrigger('cronJob')
      .timeBased()
      .everyMinutes(1)
      .create();
    
    Logger.log('✅ Cron Trigger 생성 완료 (1분마다 자동 실행)');
    return '✅ Cron Trigger 생성 완료';
  } catch (error) {
    Logger.log('❌ Trigger 생성 오류: ' + error);
    throw error;
  }
}

/**
 * Cron Trigger 삭제
 */
function deleteCronTrigger() {
  try {
    const triggers = ScriptApp.getProjectTriggers();
    let deletedCount = 0;
    
    triggers.forEach(trigger => {
      if (trigger.getHandlerFunction() === 'cronJob') {
        ScriptApp.deleteTrigger(trigger);
        deletedCount++;
      }
    });
    
    Logger.log(`✅ Cron Trigger ${deletedCount}개 삭제 완료`);
    return `✅ Cron Trigger ${deletedCount}개 삭제 완료`;
  } catch (error) {
    Logger.log('❌ Trigger 삭제 오류: ' + error);
    throw error;
  }
}

/**
 * 통계 조회
 */
function getLeadStats() {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const leadSheet = ss.getSheetByName(LEAD_SHEET_NAME);
    
    if (!leadSheet) {
      return { error: '리드데이터 시트를 찾을 수 없습니다.' };
    }
    
    const lastRow = leadSheet.getLastRow();
    if (lastRow <= 1) {
      return {
        total: 0,
        success: 0,
        pending: 0,
        error: 0
      };
    }
    
    const statuses = leadSheet.getRange(2, 6, lastRow - 1, 1).getValues();
    
    let successCount = 0;
    let pendingCount = 0;
    let errorCount = 0;
    
    statuses.forEach(row => {
      const status = String(row[0] || '').trim().toLowerCase();
      if (status === 'success') successCount++;
      else if (status === '' || status === 'pending') pendingCount++;
      else errorCount++;
    });
    
    const stats = {
      total: lastRow - 1,
      success: successCount,
      pending: pendingCount,
      error: errorCount
    };
    
    Logger.log('📊 리드 통계: ' + JSON.stringify(stats, null, 2));
    return stats;
    
  } catch (error) {
    Logger.log('❌ 통계 조회 오류: ' + error);
    return { error: error.toString() };
  }
}
