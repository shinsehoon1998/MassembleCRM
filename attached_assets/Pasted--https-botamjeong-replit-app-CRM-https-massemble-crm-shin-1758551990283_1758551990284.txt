/** 웹앱 POST */
function doPost(e) {
  try {
    Logger.log('🔥 doPost 호출됨');
    if (!e || !e.postData || !e.postData.contents) {
      return ContentService.createTextOutput(JSON.stringify({success:false,error:'데이터가 없습니다.'}))
        .setMimeType(ContentService.MimeType.JSON);
    }

    const data = JSON.parse(e.postData.contents);
    const { action, payload } = data || {};
    Logger.log('🎯 액션: ' + action);

    let result = { success:false, message:'알 수 없는 액션입니다.' };

    switch (action) {
      case 'survey_complete':      result = handleSurveyComplete(payload); break;
      case 'phone_inquiry':        result = handlePhoneInquiry(payload);   break;
      case 'consultation_request': result = handleConsultationRequest(payload); break;
      case 'get_stats':            result = getStatistics();               break;
      case 'send_crm':             result = sendToMassembleCRM(payload?.surveyId || null); break;
      case 'send_all_crm':         result = sendAllToMassembleCRM();        break;
      case 'check_crm':            result = checkMassembleCRMStatus();      break;
      default:                     result = { success:false, message:'지원하지 않는 액션: ' + action };
    }

    return ContentService.createTextOutput(JSON.stringify(result))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (error) {
    Logger.log('❌ doPost 오류: ' + error + '\n' + error.stack);
    return ContentService.createTextOutput(JSON.stringify({success:false,error:'서버 오류: ' + error.message}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

/** 웹앱 GET */
function doGet(e) {
  try {
    const action = e.parameter.action || 'stats';
    let result = {};
    if (action === 'stats')        result = getStatistics();
    else if (action === 'recent')  result = getRecentData(parseInt(e.parameter.limit) || 20);
    else if (action === 'export')  result = exportData();
    else                           result = { error: '알 수 없는 액션: ' + action };

    return ContentService.createTextOutput(JSON.stringify(result, null, 2))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({ error: '조회 오류: ' + error.message }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

/** 설문 완료 저장 + 즉시 마셈블 CRM 전송 */
function handleSurveyComplete(payload) {
  try {
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    const surveySheet = ss.getSheetByName(SHEETS.SURVEY);
    const analysisSheet = ss.getSheetByName(SHEETS.ANALYSIS);
    if (!surveySheet || !analysisSheet) throw new Error('시트를 찾을 수 없습니다. 먼저 setupInitialSheet() 실행');

    const now = new Date();
    const surveyId = payload.surveyId || generateId();

    const surveyRow = [
      now, surveyId, payload.hospitalVisits || '', payload.gender || '', payload.region || '',
      payload.premiumRange || '', payload.birthDate || '',
      Array.isArray(payload.insuranceTypes) ? payload.insuranceTypes.join(', ') : (payload.insuranceTypes || ''),
      payload.name || '', payload.phone || '', payload.consultationTime || '',
      payload.analysis?.score || 0, payload.analysis?.peerAverage || 0,
      'pending', '', '' // CRM전송상태, CRM전송시간, 마셈블고객ID
    ];
    surveySheet.appendRow(surveyRow);

    if (payload.analysis) {
      const r = payload.analysis.recommendations || [];
      const analysisRow = [
        now, surveyId, payload.analysis.score || 0, payload.analysis.peerAverage || 0,
        payload.analysis.difference || 0, payload.analysis.age || 0, payload.analysis.ageGroup || '',
        r[0] || '', r[1] || '', r[2] || '', r[3] || ''
      ];
      analysisSheet.appendRow(analysisRow);
    }

    const last = surveySheet.getLastRow();
    surveySheet.getRange(last,1).setNumberFormat('yyyy-mm-dd hh:mm:ss');
    surveySheet.getRange(last,12,1,2).setNumberFormat('#,##0');
    if (payload.analysis) {
      const la = analysisSheet.getLastRow();
      analysisSheet.getRange(la,1).setNumberFormat('yyyy-mm-dd hh:mm:ss');
      analysisSheet.getRange(la,3,1,3).setNumberFormat('#,##0');
    }

    // 즉시 마셈블 CRM 전송
    try { 
      const crmResult = sendToMassembleCRM(surveyId);
      Logger.log('🔄 CRM 자동 전송 결과: ' + JSON.stringify(crmResult)); 
    } catch (e) { 
      Logger.log('❌ 마셈블 CRM 자동 전송 실패: ' + e); 
    }

    return { success:true, message:'저장 완료', surveyId, timestamp: now.toISOString() };
  } catch (error) {
    Logger.log('❌ 설문 완료 처리 오류: ' + error);
    return { success:false, error: error.message };
  }
}

/** 마셈블 CRM으로 설문조사 데이터 전송 (특정 행 번호 기준) */
function sendToMassembleCRMByRow(rowNumber) {
  try {
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    const surveySheet = ss.getSheetByName(SHEETS.SURVEY);
    const analysisSheet = ss.getSheetByName(SHEETS.ANALYSIS);
    
    if (!surveySheet) throw new Error('설문_응답 시트를 찾을 수 없습니다.');

    const rowData = surveySheet.getRange(rowNumber, 1, 1, surveySheet.getLastColumn()).getValues()[0];
    
    // 컬럼 인덱스 (0-based)
    const IDX = { 
      responseTime: 0, id: 1, hospitalVisits: 2, gender: 3, region: 4, premiumRange: 5,
      birthDate: 6, insuranceTypes: 7, name: 8, phone: 9, consultationTime: 10,
      score: 11, peerAverage: 12, crmStatus: 13, crmSentTime: 14, massembleCustomerId: 15
    };

    // 이미 전송 성공한 경우 스킵
    let status = String(rowData[IDX.crmStatus] || '').trim().toLowerCase();
    if (status === 'success' || status === 'sending') {
      return { success: false, message: `이미 처리됨: ${status}`, row: rowNumber };
    }

    // 전송 중 상태로 변경
    surveySheet.getRange(rowNumber, IDX.crmStatus + 1).setValue('sending');

    const surveyId = rowData[IDX.id] || '';
    
    // 분석 데이터 조회
    let analysisData = null;
    if (analysisSheet && surveyId) {
      const analysisRows = analysisSheet.getDataRange().getValues().slice(1);
      analysisData = analysisRows.find(row => row[1] === surveyId); // ID 기준으로 찾기
    }

    // 전화번호 정제
    let phone = String(rowData[IDX.phone] || '').replace(/^p:/i,'').replace(/^\+82/,'0').replace(/[^0-9]/g,'');
    if (phone.length === 10 && phone.startsWith('10')) {
      phone = '0' + phone;
      surveySheet.getRange(rowNumber, IDX.phone + 1).setValue(phone); // 정제된 번호로 업데이트
    }

    // 필수 필드 검증 및 기본값 설정
   
   // 기존 코드를 다음과 같이 교체:
const crmPayload = {
  name: name || '미입력',
  phone: phone || '',
  gender: mapGender(rowData[IDX.gender]) || 'N',
  birthDate: (formatBirthDate(rowData[IDX.birthDate]) || ''),  // 추가 보호
  consultType: '보험상담',
  consultPath: '보탐정설문',
  source: 'botamjeong_survey',
  marketingConsent: false,
  
  // 모든 info 필드에 강력한 null 보호 추가
  info1: (rowData[IDX.hospitalVisits] || '') + '',
  info2: (rowData[IDX.gender] || '') + '',  
  info3: (rowData[IDX.region] || '') + '',
  info4: (rowData[IDX.premiumRange] || '') + '',
  info5: (rowData[IDX.birthDate] || '') + '',
  info6: (rowData[IDX.insuranceTypes] || '') + '',
  info7: (rowData[IDX.consultationTime] || '') + '',
  info8: (rowData[IDX.score] || '') + '',
  info9: (rowData[IDX.peerAverage] || '') + '',
  info10: `[${new Date().toLocaleString('ko-KR')}] 보탐정 설문 연동 (ID: ${surveyId})`,
  
  memo1: '',
  surveyId: surveyId || '',
  surveyCompletedAt: rowData[IDX.responseTime] ? new Date(rowData[IDX.responseTime]).toISOString() : new Date().toISOString()
};
    Logger.log('📤 마셈블 CRM 전송 데이터 (Row ' + rowNumber + '): ' + JSON.stringify(crmPayload, null, 2));

    // 마셈블 CRM API 호출
    const response = UrlFetchApp.fetch(`${MASSEMBLE_CRM_URL}/api/survey/import`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${MASSEMBLE_API_KEY}`,
        'X-API-Key': MASSEMBLE_API_KEY
      },
      payload: JSON.stringify(crmPayload),
      muteHttpExceptions: true
    });

    const responseText = response.getContentText();
    const statusCode = response.getResponseCode();
    const now = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'yyyy-MM-dd HH:mm:ss');
    
    Logger.log(`📥 마셈블 CRM 응답 (Row ${rowNumber}, ${statusCode}): ${responseText}`);

    if (statusCode === 200) {
      const result = JSON.parse(responseText);
      
      if (result.success) {
        // 성공 상태 업데이트
        surveySheet.getRange(rowNumber, IDX.crmStatus + 1).setValue('success');
        surveySheet.getRange(rowNumber, IDX.crmSentTime + 1).setValue(now);
        surveySheet.getRange(rowNumber, IDX.massembleCustomerId + 1).setValue(result.customerId || '');
        
        Logger.log(`✅ 마셈블 CRM 전송 성공 (Row ${rowNumber}): ${surveyId} → ${result.customerId}`);
        
        return {
          success: true,
          message: '마셈블 CRM 전송 성공',
          row: rowNumber,
          surveyId: surveyId,
          customerId: result.customerId,
          isNewCustomer: result.isNewCustomer,
          name: name,
          phone: phone
        };
      } else {
        // API는 성공했지만 비즈니스 로직 실패
        surveySheet.getRange(rowNumber, IDX.crmStatus + 1).setValue('api_error');
        surveySheet.getRange(rowNumber, IDX.crmSentTime + 1).setValue(now);
        
        return {
          success: false,
          message: `마셈블 CRM API 오류: ${result.message}`,
          error: result.message,
          row: rowNumber
        };
      }
    } else {
      // HTTP 오류
      surveySheet.getRange(rowNumber, IDX.crmStatus + 1).setValue('http_error');
      surveySheet.getRange(rowNumber, IDX.crmSentTime + 1).setValue(now);
      
      return {
        success: false,
        message: `마셈블 CRM HTTP 오류 (${statusCode}): ${responseText}`,
        error: `HTTP ${statusCode}`,
        row: rowNumber
      };
    }

  } catch (error) {
    Logger.log('❌ 마셈블 CRM 전송 오류 (Row ' + rowNumber + '): ' + error);
    
    // 에러 상태 업데이트
    try {
      const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
      const surveySheet = ss.getSheetByName(SHEETS.SURVEY);
      const now = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'yyyy-MM-dd HH:mm:ss');
      surveySheet.getRange(rowNumber, 14).setValue('error'); // CRM전송상태
      surveySheet.getRange(rowNumber, 15).setValue(now); // CRM전송시간
    } catch (e) {
      Logger.log('상태 업데이트 실패: ' + e);
    }
    
    return {
      success: false,
      message: '마셈블 CRM 전송 중 오류 발생: ' + error.message,
      error: error.message,
      row: rowNumber
    };
  }
}

/** surveyId로(있으면) 또는 마지막 행으로 마셈블 CRM 전송 */
function sendToMassembleCRM(targetSurveyId = null) {
  try {
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    const sheet = ss.getSheetByName(SHEETS.SURVEY);
    if (!sheet) throw new Error('설문_응답 시트를 찾을 수 없습니다.');

    const lastRow = sheet.getLastRow();
    if (lastRow <= 1) return { success: false, message: '전송할 데이터가 없습니다.' };

    if (targetSurveyId) {
      // 특정 surveyId 찾기
      const allData = sheet.getDataRange().getValues();
      for (let i = 1; i < allData.length; i++) {
        if (allData[i][1] === targetSurveyId) { // ID 컬럼
          return sendToMassembleCRMByRow(i + 1); // 1-based row number
        }
      }
      throw new Error(`설문 ID ${targetSurveyId}를 찾을 수 없습니다.`);
    } else {
      // 마지막 행 전송
      return sendToMassembleCRMByRow(lastRow);
    }
  } catch (e) {
    return { success: false, message: `마셈블 CRM 전송 오류: ${e.message}` };
  }
}

/** 모든 미전송 데이터를 마셈블 CRM으로 일괄 전송 */
function sendAllToMassembleCRM() {
  try {
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    const sheet = ss.getSheetByName(SHEETS.SURVEY);
    if (!sheet) throw new Error('설문_응답 시트를 찾을 수 없습니다.');

    const data = sheet.getDataRange().getValues();
    const headers = data[0];
    const rows = data.slice(1);

    const statusIdx = headers.indexOf('CRM전송상태');
    if (statusIdx === -1) throw new Error('CRM전송상태 컬럼을 찾을 수 없습니다. setupInitialSheet() 실행 필요');

    let successCount = 0;
    let failCount = 0;
    const results = [];

    for (let i = 0; i < rows.length; i++) {
      const row = rows[i];
      const status = String(row[statusIdx] || '').trim().toLowerCase();
      const surveyId = row[1]; // ID 컬럼

      // pending, 빈 상태, 또는 실패 상태인 항목만 전송
      if (status === 'pending' || !status || status === 'fail' || status === 'error' || status === 'api_error' || status === 'http_error') {
        try {
          const result = sendToMassembleCRMByRow(i + 2); // +2: 헤더 행 때문에
          results.push({ 
            row: i + 2, 
            surveyId: surveyId, 
            result: result 
          });
          
          if (result.success) {
            successCount++;
          } else {
            failCount++;
          }
          
          // API 호출 간격 (Rate Limiting 방지)
          Utilities.sleep(1000);
          
        } catch (error) {
          failCount++;
          results.push({
            row: i + 2,
            surveyId: surveyId,
            result: {
              success: false,
              error: error.message
            }
          });
        }
      }
    }

    return {
      success: true,

      successCount: successCount,
      failCount: failCount,
      results: results
    };

  } catch (error) {
    return {
      success: false,
      message: '마셈블 CRM 일괄 전송 중 오류 발생: ' + error.message,
      error: error.message
    };
  }
}

/** 마셈블 CRM 연동 상태 확인 */
function checkMassembleCRMStatus() {
  try {
    // 1. 마셈블 CRM 서버 상태 확인
    const response = UrlFetchApp.fetch(`${MASSEMBLE_CRM_URL}/api/survey/status`, {
      method: 'GET',
      headers: {
        'Authorization': `Bearer ${MASSEMBLE_API_KEY}`,
        'X-API-Key': MASSEMBLE_API_KEY
      },
      muteHttpExceptions: true
    });

    const statusCode = response.getResponseCode();
    const responseText = response.getContentText();

    let serverStatus = {};
    if (statusCode === 200) {
      const result = JSON.parse(responseText);
      serverStatus = {
        connected: true,
        status: result.status || 'operational',
        recentCustomers: result.recentCustomers || 0,
        message: result.message || '마셈블 CRM 연동 정상'
      };
    } else {
      serverStatus = {
        connected: false,
        error: `HTTP ${statusCode}`,
        message: `마셈블 CRM 연결 실패 (${statusCode}): ${responseText}`
      };
    }

    // 2. 로컬 스프레드시트 전송 상태 집계
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    const sheet = ss.getSheetByName(SHEETS.SURVEY);
    if (!sheet) throw new Error('설문_응답 시트를 찾을 수 없습니다.');

    const data = sheet.getDataRange().getValues();
    const headers = data[0];
    const rows = data.slice(1);
    const statusIdx = headers.indexOf('CRM전송상태');
    
    let localStatus = { total: 0, success: 0, pending: 0, failed: 0 };
    
    if (statusIdx !== -1) {
      localStatus.total = rows.length;
      
      rows.forEach(row => {
        const status = String(row[statusIdx] || '').trim().toLowerCase();
        if (status === 'success') {
          localStatus.success++;
        } else if (status === 'pending' || !status) {
          localStatus.pending++;
        } else {
          localStatus.failed++;
        }
      });
    }

    localStatus.successRate = localStatus.total ? Math.round((localStatus.success / localStatus.total) * 100) : 0;

    return {
      success: true,
      server: serverStatus,
      local: localStatus,
      summary: {
        isHealthy: serverStatus.connected && localStatus.successRate > 80,
        pendingCount: localStatus.pending,
        failedCount: localStatus.failed
      }
    };

  } catch (error) {
    return {
      success: false,
      message: '마셈블 CRM 상태 확인 오류: ' + error.message,
      error: error.message
    };
  }
}

/** 성별 매핑 함수 */
function mapGender(gender) {
  if (!gender) return 'N';
  const g = String(gender).toLowerCase();
  if (g.includes('남') || g.includes('male') || g === 'm') return 'M';
  if (g.includes('여') || g.includes('female') || g === 'f') return 'F';
  return 'N';
}

/** 생년월일 포맷 함수 - null 값 방지 */
function formatBirthDate(birthDate) {
  if (!birthDate) return '';  // null 대신 빈 문자열 반환
  
  try {
    let dateStr = String(birthDate);
    
    // 이미 Date 객체인 경우
    if (birthDate instanceof Date) {
      return Utilities.formatDate(birthDate, Session.getScriptTimeZone(), 'yyyy-MM-dd');
    }
    
    // YYYY-MM-DD 형식은 그대로 반환
    if (/^\d{4}-\d{2}-\d{2}$/.test(dateStr)) {
      return dateStr;
    }
    
    // YYYYMMDD 형식 처리
    if (/^\d{8}$/.test(dateStr)) {
      return `${dateStr.substring(0,4)}-${dateStr.substring(4,6)}-${dateStr.substring(6,8)}`;
    }
    
    // YYYY.MM.DD 형식 처리
    if (/^\d{4}\.\d{2}\.\d{2}$/.test(dateStr)) {
      return dateStr.replace(/\./g, '-');
    }
    
    // YYYY/MM/DD 형식 처리
    if (/^\d{4}\/\d{2}\/\d{2}$/.test(dateStr)) {
      return dateStr.replace(/\//g, '-');
    }
    
    return '';  // 파싱 실패 시 빈 문자열 반환
  } catch (e) {
    Logger.log('생년월일 포맷 오류: ' + e);
    return '';  // 오류 시 빈 문자열 반환
  }
}

    };

    if (surveySheet && surveySheet.getLastRow() > 1) {
      const data = surveySheet.getDataRange().getValues();
      const headers = data[0];
      const statusIdx = headers.indexOf('CRM전송상태');
      
      if (statusIdx !== -1) {
        const rows = data.slice(1);
        stats.crmStatus = { success: 0, pending: 0, failed: 0 };
        
        rows.forEach(row => {
          const status = String(row[statusIdx] || '').trim().toLowerCase();
          if (status === 'success') stats.crmStatus.success++;
          else if (status === 'pending' || !status) stats.crmStatus.pending++;
          else stats.crmStatus.failed++;
        });

/** 최근 데이터 조회 */
function getRecentData(limit = 20) {
  try {
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    const sheet = ss.getSheetByName(SHEETS.SURVEY);
    if (!sheet) throw new Error('설문_응답 시트를 찾을 수 없습니다.');
    
    const lastRow = sheet.getLastRow();
    if (lastRow <= 1) return { success: true, data: [] };
    
    const startRow = Math.max(2, lastRow - limit + 1);
    const range = sheet.getRange(startRow, 1, lastRow - startRow + 1, sheet.getLastColumn());
    const values = range.getValues();
    const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
    
    const data = values.map(row => {
      const obj = {};
      headers.forEach((header, index) => {
        obj[header] = row[index];
      });
      return obj;
    }).reverse(); // 최신 순으로 정렬
    
    return { success: true, data: data };
  } catch (error) {
    return { success: false, error: error.message };
  }
}

/** 1분 주기 크론 트리거 생성 */
function createCronTrigger() {
  try {
    // 기존 트리거 삭제
    ScriptApp.getProjectTriggers().forEach(trigger => {
      if (trigger.getHandlerFunction() === 'cronJob') {
        ScriptApp.deleteTrigger(trigger);
      }
    });
    
    // 새 트리거 생성 (1분마다)
    ScriptApp.newTrigger('cronJob')
      .timeBased()
      .everyMinutes(1)
      .create();
    
    Logger.log('✅ 크론 트리거 생성됨 (1분 주기)');
    return '크론 트리거 생성 완료';
  } catch (e) {
    Logger.log('❌ 크론 트리거 생성 오류: ' + e);
    throw e;
  }
}

/** 크론 잡 실행 함수 */
function cronJob() {
  try {
    Logger.log('⏰ 크론 잡 실행 시작');
    
    // 미전송 데이터가 있는지 확인
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    const sheet = ss.getSheetByName(SHEETS.SURVEY);
    
    if (!sheet || sheet.getLastRow() <= 1) {
      Logger.log('📭 전송할 데이터가 없음');
      return;
    }
    
    const data = sheet.getDataRange().getValues();
    const headers = data[0];
    const statusIdx = headers.indexOf('CRM전송상태');
    
    if (statusIdx === -1) {
      Logger.log('❌ CRM전송상태 컬럼을 찾을 수 없음');
      return;
    }
    
    const rows = data.slice(1);
    let pendingCount = 0;
    
    rows.forEach(row => {
      const status = String(row[statusIdx] || '').trim().toLowerCase();
      if (status === 'pending' || !status) {
        pendingCount++;
      }
    });
    
    if (pendingCount > 0) {
      Logger.log(`📤 미전송 데이터 ${pendingCount}건 발견. 일괄 전송 시작`);
      const result = sendAllToMassembleCRM();
      Logger.log('📤 크론 잡 일괄 전송 결과: ' + JSON.stringify(result));
    } else {
      Logger.log('✅ 모든 데이터가 이미 전송됨');
    }
    
  } catch (error) {
    Logger.log('❌ 크론 잡 오류: ' + error);
  }
}
